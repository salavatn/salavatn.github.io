Simple Example

from faker import Faker

fake = Faker()

name = fake.name()
country = fake.country()

print(f"{name} from {country}!")

Example of using Faker

from faker import Faker

fake = Faker()

name = fake.first_name_male()
time = fake.time()
month = fake.month_name()
delivery_date = fake.future_datetime()
domain = fake.domain_name()
person = fake.name()
email = fake.free_email()
phone = fake.phone_number()

msg = f"Mr. {name}, \n" \
      f"Your package has been shipped - ETA: {time} \n" \
      f"on 18th {month} at your residence. \n" \
      f"Estimated delivery date is {delivery_date}\n" \
      f"To track your parcel, visit www.{domain}.\n" \
      f"\n" \
      f"Best regards,\n" \
      f"{person}\n" \
      f"Email:\t{email}\n" \
      f"Phone:\t{phone}"


print(msg)
